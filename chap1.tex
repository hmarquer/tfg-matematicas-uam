\chapter{Geometría de la clase de Schur}\label{chap1}

\begin{defn}\label{defn:clase-schur}
    Definimos la \textit{clase de Schur} como $\mathscr{S} \defeq \left\{ f \colon \mathbb{D} \longrightarrow \bar{\mathbb{D}} : f \text{ es \exhyperref[defn:fn-holomorfa]{fn-holomorfa}{holomorfa}} \right\}$.
\end{defn}

\begin{lem}[de Schwarz]\label{lem:schwarz}
    Sea $f \in \mathscr{S}$ tal que $f(0) = 0$, entonces
    \begin{enu}[2][label=\textnormal{(\arabic*)}, ref=\textnormal{(\arabic*)}]
        \item\label{lem:schwarz:1} $\forall z \in \mathbb{D} : \abs{f(z)} \leq \abs{z}$.
        \item\label{lem:schwarz:2} $\abs{f\exhyperref[defn:fn-compleja-derivable-pnt]{fn-compleja-derivable-pnt}{'}(0)} \leq 1$.
    \end{enu}

    Además, si se cumple la igualdad en \ref{lem:schwarz:1} para $z \neq 0$ o en \ref{lem:schwarz:2}, $f$ es una \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{rotación}.
\end{lem}

\section{Automorfismos del disco unidad}

\begin{defn}[Automorfismo del disco unidad]\label{defn:automorfismo-disco-unidad}
    Para $f \colon \mathbb{D} \to \mathbb{D}$, decimos que $f$ es un \emph{automorfismo del disco unidad} si es biyectiva y \exhyperref[defn:fn-holomorfa]{fn-holomorfa}{holomorfa}. El conjunto de automorfismos del disco unidad se denota por $\mathrm{Aut} \left(\mathbb{D} \right)$.
\end{defn}

\begin{ejem}[Rotación]\label{ejem:rotacion-disco-unidad}
    Sea $\gamma \in \mathbb{T}$, entonces $\rho_\gamma(z) \defeq \gamma z \implies \rho_\gamma \in \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{\mathrm{Aut} \left(\mathbb{D}\right)}$.
\end{ejem}

\begin{lem}\label{lem:aut-disco-unidad-grupo}
    \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{$\mathrm{Aut} \left(\mathbb{D} \right)$} es un \exhyperref[defn:grupo]{grupo}{grupo} bajo la composición de funciones.
\end{lem}
\begin{dem}
    % Comprobamos las propiedades de la definición de grupo:
    \begin{enumerate}[label=(\roman*), wide, labelindent=0pt]
        \item Asociatividad: $\forall f, g, h \in \mathrm{Aut} \left(\mathbb{D} \right) : (f \circ g) \circ h = f \circ (g \circ h)$.
        \item Elemento neutro: $\operatorname{id} \in \mathrm{Aut} \left(\mathbb{D} \right)$ cumple que $\forall f \in \mathrm{Aut} \left(\mathbb{D} \right) : f \circ \operatorname{id} = f = \operatorname{id} \circ f$.
        \item Elemento inverso: $\forall f \in \mathrm{Aut} \left(\mathbb{D} \right) : f^{-1}$ satisface $f \circ f^{-1} = \operatorname{id} = f^{-1} \circ f$. Veamos que $f^{-1} \in \mathrm{Aut} \left(\mathbb{D} \right)$. Como $f$ es biyectiva, en particular es inyectiva y al aplicar el \excref[lem:localmente-inyectiva-implica-inversa-no-nula]{lem-localmente-inyectiva-implica-inversa-no-nula}, obtenemos que $\forall z \in \mathbb{D} : f'(z) \neq 0$. Entonces, por el \exhyperref[teo:fn-inversa-holomorfas]{teo-fn-inversa-holomorfas}{teorema de la función inversa}, $f^{-1} \in \mathcal{H}(\mathbb{D})$, luego $f^{-1} \in \mathrm{Aut} \left(\mathbb{D} \right)$ ya que $f^{-1}$ es biyectiva.\hfill\qedsymbol
    \end{enumerate}
\end{dem}

\subsection{Involuciones del disco unidad}

\begin{defn}[Involución del disco unidad]\label{defn:involucion-disco-unidad}
    Sea $w \in \mathbb{D}$. A la aplicación $\tau_w \colon \mathbb{D} \to \mathbb{C}$ dada por $\tau_w (z) \defeq \frac{w - z}{1 - \bar{w} z}$ la llamaremos \emph{involución del disco unidad} (asociada a $w$).
\end{defn}

\begin{lem}\label{lem:involucion-disco-unidad}
    Sea $w \in \mathbb{D}$ y $\tau_w$ la \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{involución del disco unidad asociada a $w$}, entonces
    \begin{enu}[2][label=\textnormal{(\arabic*)}, ref=\textnormal{(\arabic*)}]
        \item\label{lem:involucion-disco-unidad:1} $z \in \mathbb{D} \iff \abs{\tau_w (z)} < 1$.
        \item\label{lem:involucion-disco-unidad:2} $z \in \mathbb{T} \iff \abs{\tau_w (z)} = 1$.
        \item\label{lem:involucion-disco-unidad:3} $\tau_w \in \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{\mathrm{Aut} \left(\mathbb{D} \right)}$.
        \item\label{lem:involucion-disco-unidad:4} $\tau_w \circ \tau_w = \operatorname{id}$.
    \end{enu}
\end{lem}
\begin{dem}
    \begin{enumerate}[label=\textnormal{(\arabic*)}, wide, labelindent=0pt]
        \item Usaremos que $\abs{z_1 + z_2}^2 = \abs{z_1}^2 + 2\, \Re\left(\bar{z_1} z_2\right) + \abs{z_2}^2$ para todo $z_1, z_2 \in \C$. Sea $z \in \C$, entonces $\abs{\tau_w(z)} < 1$ si y solo si
        \[
        \begin{aligned}
            {\abs{\frac{w-z}{1- \bar{w} z}}}^2 < 1 &\iff \abs{w}^2 - \cancel{2\, \Re\left(\bar{w} z\right)} + \abs{z}^2 < 1 - \cancel{2\, \Re\left(\bar{w} z\right)} + \abs{w}^2 \abs{z}^2 \\
            &\iff 1 + \abs{w}^2 \abs{z}^2 - \abs{w}^2 - \abs{z}^2 > 0 \iff \left(1 - \abs{w}^2\right) \left(1- \abs{z}^2\right) > 0 \\
            &\iff 1 - \abs{z}^2 > 0 \iff \abs{z} < 1 \iff z \in \mathbb{D}. 
        \end{aligned}
        \]

        \item El mismo cálculo de \ref{lem:involucion-disco-unidad:1} nos da que $\abs{\tau_w(z)} = 1 \iff z \in \mathbb{T}$.

        \item Por \ref{lem:involucion-disco-unidad:1}, $\tau_w(\mathbb{D}) = \mathbb{D}$ y por \ref{lem:involucion-disco-unidad:2}, $\tau_w(\mathbb{T}) = \mathbb{T}$. Como además $\tau_w$ es una \exhyperref[defn:transformacion-mobius]{transformacion-mobius}{transformación de Möbius}, es \exhyperref[defn:fn-holomorfa]{fn-holomorfa}{holomorfa} e inyectiva, por lo que $\tau_w \in \mathrm{Aut} \left(\mathbb{D}\right)$.

        \item $\ds \tau_w \circ \tau_w (z) = \frac{w - \frac{w - z}{1 - \bar{w} z}}{1 - \bar{w} \cdot \frac{w - z}{1 - \bar{w} z}} = \frac{w(1 - \bar{w} z) - (w - z)}{(1 - \bar{w} z) - \bar{w}(w - z)} = \frac{z - \abs{w}^2 z}{1 - \abs{w}^2} = z$.\hfill\qedsymbol
    \end{enumerate}
\end{dem}

\begin{obs}\label{obs:involucion-disco-unidad-diametro}
    Sea $w \in \mathbb{D}$, entonces $\Gamma_w \defeq \left\{t \, w/\abs{w} : t  \in (-1,1)\right\} \subset \mathbb{D}$ es el diámetro del disco unidad en la dirección de $w$. Al aplicar $\tau_w$, obtenemos que
    \[
    \forall t \in (-1,1) : \tau_w\left(t \, \frac{w}{\abs{w}}\right) = \frac{w - t \, w/\abs{w}}{1 - \bar{w} t \, w/\abs{w}} = \frac{\abs{w} - t}{1 - t \abs{w}} \cdot \frac{w}{\abs{w}} \in \Gamma_w
    \]
    porque $\frac{\abs{w} - t}{1 - t \abs{w}} \in \R$ y $\tau_w(\mathbb{D}) \subset \mathbb{D}$. Por tanto, $\tau_w \left(\Gamma_w\right) \subset \Gamma_w$ y, aplicando el apartado \ref{lem:involucion-disco-unidad:4} del \excref[lem:involucion-disco-unidad]{lem-involucion-disco-unidad}, $\Gamma_w = \tau_w \circ \tau_w \left(\Gamma_w\right) \subset \tau_w \left(\Gamma_w\right)$. En consecuencia, $\tau_w \left(\Gamma_w\right) = \Gamma_w$. 
\end{obs}

\begin{lem}\label{lem:involucion-disco-unidad-derivada}
    Sea $w \in \mathbb{D}$, entonces $\ds \forall z \in \mathbb{D} : \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_w}\exhyperref[defn:fn-compleja-derivable-pnt]{fn-compleja-derivable-pnt}{'}(z) = \frac{\abs{w}^2 - 1}{(1 - \bar{w} z)^2}$.
\end{lem}
\begin{dem}
    Usando la \exhyperref[prop:fn-compleja-derivable]{prop-fn-compleja-derivable}{regla del cociente}, tenemos que para $z \in \mathbb{D}$,
    \[
    {\tau_w}'(z) = \frac{-1 \cdot \left(1 - \bar{w} z\right) - (w - z) \cdot (-\bar{w})}{\left(1 - \bar{w} z\right)^2} = \frac{\cancel{\bar{w} z} - 1 + \bar{w} w - \cancel{\bar{w} z}}{\left(1 - \bar{w} z\right)^2} = \frac{\abs{w}^2 - 1}{(1 - \bar{w} z)^2}.
    \]

    \vspace{-\baselineskip}
    \vspace{-\belowdisplayskip}
    \hfill\qedsymbol
\end{dem}

\subsection{Caracterización de $\mathrm{Aut} \left(\mathbb{D} \right)$}

El siguiente teorema nos da una caracterización de los automorfismos del disco unidad mediante una fórmula explícita.

\begin{teo}\label{teo:formula-aut-disco-unidad}
    Sea $\ds f \in \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{\mathrm{Aut} \left(\mathbb{D}\right)}$, entonces $\exists !\, \gamma \in \mathbb{T}, w \in \mathbb{D} : f = \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_\gamma} \circ \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_w}$.

    En particular, los \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{automorfismos del disco unidad} son \exhyperref[defn:transformacion-mobius]{transformacion-mobius}{transformaciones de Möbius}.
\end{teo}
\begin{dem}
    Definimos $w \defeq f^{-1}(0) \in \mathbb{D}$ y $h \defeq f \circ \tau_w \in \mathcal{H}\left(\mathbb{D}\right)$. Entonces $h(0) = f(w) = 0$ y $h^{-1}(0) = \tau_w(w) = 0$, por lo que estamos en las condiciones del \exhyperref[lem:schwarz]{lem-schwarz}{lema de Schwarz} tanto para $h$ como para $h^{-1} \in \mathcal{H}\left(\mathbb{D}\right)$.
    Por tanto,
    \[
    \forall z \in \mathbb{D} : \abs{h(z)} \leq \abs{z} \we \abs{h^{-1}(z)} \leq \abs{z}.
    \]
    De la segunda desigualdad, obtenemos que $\abs{z} = \abs{h^{-1}(h(z))} \leq \abs{h(z)}$ y, con la primera, deducimos que $\abs{h(z)} = \abs{z}$. Entonces, el \exhyperref[lem:schwarz]{lem-schwarz}{lema de Schwarz} garantiza que $h$ es una rotación, i.e. $\exists\, \gamma \in \mathbb{T} : h = \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_\gamma}$. Aplicando el apartado \exref[lem:involucion-disco-unidad:4]{lem-involucion-disco-unidad} del \excref[lem:involucion-disco-unidad]{lem-involucion-disco-unidad}, despejamos $f$:
    \[
    h = f \circ \tau_w \implies h \circ \tau_w = f \circ \tau_w \circ \tau_w \implies f = \rho_\gamma \circ \tau_w.
    \]
    Como $w$ es único por ser $f$ biyectiva, $\gamma$ también es único.\hfill\qedsymbol
\end{dem}

\begin{obs}\label{obs:aut-disco-unidad-fija-origen-imp-rotacion}
    Del \excref[teo:formula-aut-disco-unidad]{teo-formula-aut-disco-unidad}, deducimos que si $f \in \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{\mathrm{Aut} \left(\mathbb{D} \right)}$ es tal que $f(0) = 0$, entonces $f = \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_\gamma}$ para algún $\gamma \in \mathbb{T}$. Es decir, los únicos \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{automorfismos del disco unidad} que fijan el origen son las rotaciones.
\end{obs}

\begin{prop}\label{prop:aut-disco-unidad-traslada-z1-0-z2-01}
    Sean $z_1, z_2 \in \mathbb{D}$ tales que $z_1 \neq z_2$, entonces existe $f \in \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{\mathrm{Aut} \left(\mathbb{D} \right)}$ tal que $f(z_1) = 0$ y $f(z_2) \in (0,1) \subset \mathbb{R}$.
\end{prop}
\begin{dem}
    Consideramos $g \defeq f \circ \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_{z_1}}$, entonces $g \in \mathrm{Aut} \left(\mathbb{D} \right)$ y $g(0) = f(z_1) = 0$. Por la \excref[obs:aut-disco-unidad-fija-origen-imp-rotacion]{obs-aut-disco-unidad-fija-origen-imp-rotacion}, $\exists \gamma \in \mathbb{T} : g = \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_\gamma}$. Por tanto, $f = g \circ \tau_{z_1}^{-1} = \rho_\gamma \circ \tau_{z_1}$.


    Si imponemos ahora que $f(z_2) \in (0, 1)$, entonces $\rho_\gamma(\tau_{z_1}(z_2)) = \gamma \, \tau_{z_1}(z_2) \in (0, 1)$. Por tanto, $\gamma \tau_{z_1}(z_2) = \abs{\tau_{z_1}(z_2)}$, luego $\gamma = \frac{\abs{\tau_{z_1}(z_2)}}{\tau_{z_1}(z_2)} \in \mathbb{T}$. \hfill \qedsymbol
\end{dem}

\section{Estructura algebraica de $\mathrm{Aut} \left(\mathbb{D} \right)$}

\begin{teo}\label{teo:aut-disco-unidad-parametros}
    Sea $f = \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_\gamma} \circ \exhyperref[lem:involucion-disco-unidad]{lem-involucion-disco-unidad}{\tau_w} \in \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{\mathrm{Aut} \left(\mathbb{D} \right)}$, entonces
    \[
    w = f^{-1}(0) \quad \we\quad \gamma = \begin{cases}
    \sfrac{f(0)}{f^{-1}(0)} & \text{si } f(0) \neq 0, \\
    -f\exhyperref[defn:fn-compleja-derivable-pnt]{fn-compleja-derivable-pnt}{'}(0) & \text{si } f(0) = 0.
    \end{cases}
    \]
\end{teo}
\begin{dem}
    Tenemos que $f(w) = \rho_\gamma \circ \tau_w(w) = \gamma \cdot 0 = 0$, luego $w = f^{-1}(0)$.

    Por otro lado, $f(0) = \rho_\gamma \circ \tau_w(0) = \gamma \cdot w = \gamma \cdot f^{-1}(0)$. Por tanto, tenemos que
    \begin{itemize}
        \item Si $f(0) \neq 0$, entonces $\gamma = \sfrac{f(0)}{f^{-1}(0)}$.
        \item Si $f(0) = 0$, entonces, por la \excref[obs:aut-disco-unidad-fija-origen-imp-rotacion]{obs-aut-disco-unidad-fija-origen-imp-rotacion}, $\exists \zeta \in \mathbb{T} : f = \rho_\zeta$. Como $f = \rho_\gamma \circ \tau_0$ y $\tau_0 = -\operatorname{id}$, entonces $\forall z \in \mathbb{D} : f(z) = - \gamma z$, luego $\gamma = - f'(z) = -f'(0)$.\hfill\qedsymbol
    \end{itemize}
\end{dem}

\begin{lem}\label{lem:involucion-disco-unidad-rho}
    Sea $f = \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_w} \circ \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_\gamma} \in \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{\mathrm{Aut} \left(\mathbb{D} \right)}$, entonces $f = \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_\gamma} \circ \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_{\bar{\gamma} w}}$.
\end{lem}
\begin{dem}
    Como $\gamma \in \mathbb{T}$, tenemos que $\gamma \cdot \bar{\gamma} = \abs{\gamma}^2 = 1$. Por tanto, $\forall z \in \mathbb{D}:$
    \[
    f(z) = \tau_w(\rho_\gamma(z)) = \frac{w - \gamma z}{1 - \bar{w} \gamma z} \cdot \frac{\bar{\gamma}}{\bar{\gamma}} = \frac{1}{\bar{\gamma}} \cdot \frac{\bar{\gamma} w - z}{1 - \bar{w}\gamma z} = \gamma \cdot \frac{\bar{\gamma} w - z}{1 - \overline{(\bar{\gamma} w)} z} = \rho_\gamma(\tau_{\bar{\gamma} w}(z)).
    \]

    \vspace{-\baselineskip}
    \vspace{-\belowdisplayskip}
    \hfill\qedsymbol
\end{dem}

\begin{prop}\label{prop:aut-disco-unidad-inversa}
    Sea $f = \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_{\gamma}} \circ \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_w} \in \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{\mathrm{Aut} \left(\mathbb{D} \right)}$, entonces $f^{-1} = \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_{w}} \circ \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_{\bar{\gamma}}} = \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_{\bar{\gamma}}} \circ \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_{\gamma w}}$.
\end{prop}
\begin{dem}
    Por el \excref[lem:involucion-disco-unidad]{lem-involucion-disco-unidad} y la identidad $\rho_\gamma^{-1} = \rho_{\gamma^{-1}} = \rho_{\bar{\gamma}}$, obtenemos que
    \[
    \left(\rho_\gamma \circ \tau_w\right)^{-1} = \tau_w^{-1} \circ \rho_\gamma^{-1} = \tau_w \circ \rho_{\gamma}^{-1} = \tau_w \circ \rho_{\bar{\gamma}}.
    \]
    Para la segunda igualdad, usamos el \excref[lem:involucion-disco-unidad-rho]{lem-involucion-disco-unidad-rho}: $\tau_{w} \circ \rho_{\bar{\gamma}} = \rho_{\bar{\gamma}} \circ \tau_{\bar{\left(\bar{\gamma}\right)} w} = \rho_{\bar{\gamma}} \circ \tau_{\gamma w}$.
    \hfill \qedsymbol
\end{dem}

\begin{lem}\label{lem:aut-disco-unidad-composicion-tau}
    Sean $a, b \in \mathbb{D}$, entonces
    \[
    \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_b} \circ \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_a} = \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_{\gamma}} \circ \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_c} \;\;\text{ donde }\;\; c = \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_a}(b) = \frac{a - b}{1 - \bar{a} b} \;\;\we\;\; \gamma = \begin{cases}
        \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_{\bar{a}b}}(1) & \text{si } a \neq b, \\
        -1 & \text{si } a = b.
    \end{cases}
    \]
\end{lem}
\begin{dem}
    Por el \excref[teo:aut-disco-unidad-parametros]{teo-aut-disco-unidad-parametros}, sabemos que
    \[
    c = \left(\tau_b \circ \tau_a\right)^{-1}(0) = \left(\tau_a^{-1} \circ \tau_b^{-1}\right) (0) = \tau_a \left(\tau_b(0)\right) = \tau_a(b) = \frac{a - b}{1 - \bar{a} b}. 
    \]
    Si $a = b$, entonces $c = 0$ y $\tau_b \circ \tau_a = \operatorname{id} = \rho_{-1} \circ \tau_0$, luego $\gamma = -1$.
    Si $a \neq b$, entonces $\left(\tau_b \circ \tau_a\right)(0) = \rho_{\gamma}(c)$, luego $\tau_b(a) = \gamma \, \tau_a(b)$ y, como $a \neq b$, $\tau_a(b) \neq 0$, tenemos
    \[
    \gamma = \frac{\tau_b(a)}{\tau_a(b)} = \frac{\sfrac{(b - a)}{(1 - \bar{b} a)}}{\sfrac{(a - b)}{(1 - \bar{a} b)}} = \frac{\bar{a} b - 1}{1 - a \bar{b}} = \tau_{\bar{a} b}(1).
    \]

    \vspace{-\baselineskip}
    \vspace{-\belowdisplayskip}
    \hfill\qedsymbol
\end{dem}

\begin{teo}\label{teo:aut-disco-unidad-composicion}
    Sean $f = \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_{\gamma_1}} \circ \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_{a_1}}$ y $g = \exhyperref[ejem:rotacion-disco-unidad]{ejem-rotacion-disco-unidad}{\rho_{\gamma_2}} \circ \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_{a_2}}$ en $\exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{\mathrm{Aut} \left(\mathbb{D} \right)}$, entonces
    \[
    g \circ f = \rho_{\gamma} \circ \tau_a \;\;\text{ donde }\;\; a = \tau_{a_1}(a_2) = \frac{a_1 - a_2}{1 - \bar{a_1} a_2} \;\;\we\;\; \gamma = \begin{cases}
        \gamma_1 \, \tau_{\bar{a_1} a_2}(\gamma_2) & \text{si } a_2 \neq a_1 \gamma_1, \\
        -\gamma_1 \gamma_2 & \text{si } a_2 = a_1 \gamma_1,
    \end{cases}
    \]
    En particular, si $a_2 = a_1 \gamma_1$, entonces $g \circ f = \rho_{\gamma_1 \gamma_2}$.
\end{teo}
\begin{dem}
    Operando: $\smash{g \circ f =\rho_{\gamma_2} \circ \left(\tau_{a_2} \circ \rho_{\gamma_1}\right) \circ \tau_{a_1} \upexpl{=}{\exref[lem:involucion-disco-unidad-rho]{lem-involucion-disco-unidad-rho}} \rho_{\gamma_2} \circ \rho_{\gamma_1} \circ \left(\tau_{\bar{\gamma_1} a_2} \circ \tau_{a_1}\right)}$.

    Ahora bien, por el \excref[lem:aut-disco-unidad-composicion-tau]{lem-aut-disco-unidad-composicion-tau}, tenemos que $\tau_{\bar{\gamma_1} a_2} \circ \tau_{a_1} = \rho_{\widetilde{\gamma}} \circ \tau_{\widetilde{a}}$ donde
    \[
    \widetilde{\gamma} = \begin{cases}
        \tau_{\bar{a_1} a_2 \bar{\gamma_2}}(1) & \text{si } a_1 \neq \bar{\gamma_1} a_2, \\
        -1 & \text{si } a_1 = \bar{\gamma_1} a_2,
    \end{cases} \;\;\we\;\; \widetilde{a} = \tau_{a_1}(\bar{\gamma_1} a_2) = \frac{a_1 - \bar{\gamma_1} a_2}{1 - \bar{a_1} \bar{\gamma_1} a_2}.
    \]
    Concluimos por tanto que si $g \circ f = \rho_{\gamma} \circ \tau_a$, entonces
    \[
    \gamma = \gamma_1 \gamma_2 \widetilde{\gamma} = \begin{cases}
        \gamma_1 \gamma_2 \cdot \tau_{\bar{a_1} a_2 \bar{\gamma_2}}(1) & \text{si } a_2 \neq a_1 \gamma_1, \\
        -\gamma_1 \gamma_2 & \text{si } a_2 = a_1 \gamma_1,
    \end{cases} \quad \we \quad a = \widetilde{a} = \tau_{a_1}(\bar{\gamma_1} a_2).
    \]
    Finalmente, tenemos que $\ds \gamma_2 \cdot \tau_{\bar{a_1} a_2 \bar{\gamma_2}}(1) = \gamma_2 \cdot \frac{\bar{a_1} a_2 \bar{\gamma_2} - 1}{1 - a_1 \bar{a_2} \gamma_2} = \frac{\bar{a_1} a_2 - \gamma_2}{1 - a_1 \bar{a_2} \gamma_2} = \tau_{\bar{a_1} a_2}(\gamma_2)$ y se sigue el resultado deseado.\hfill\qedsymbol
\end{dem}

\subsection{$\mathrm{Aut} \left(\mathbb{D} \right)$ como subgrupo de las aplicaciones de Möbius}

Por definición, toda \exhyperref[defn:transformacion-mobius]{transformacion-mobius}{transformación de Möbius} viene dada por una matriz invertible $A \in \exhyperref[prop:grupo-lineal]{grupo-lineal}{\operatorname{GL}_2}(\mathbb{C})$. Sin embargo, $\forall \lambda \in \mathbb{C} \setminus \{0\} : A$ y $\lambda A$ definen la misma \exhyperref[defn:transformacion-mobius]{transformacion-mobius}{transformación de Möbius}. Por tanto, el \exhyperref[prop:trans-mobius-grupo]{prop-trans-mobius-grupo}{grupo de las transformaciones de Möbius} es isomorfo al \exhyperref[defn:grupo-proyectivo-lineal]{grupo-proyectivo-lineal}{grupo proyectivo lineal} $\operatorname{PGL}_2(\mathbb{C})$ dado por
\[
\operatorname{PGL}_2(\mathbb{C}) \defeq \sfrac{\operatorname{GL}_2(\mathbb{C})}{\sim} \quad \text{ donde }\quad A \sim B \iff \exists \lambda \in \mathbb{C} \setminus \{0\} : A = \lambda B. 
\]
Además, como $\C$ es un cuerpo algebraicamente cerrado, toda clase de equivalencia $[A] \in \operatorname{PGL}_2(\mathbb{C})$ tiene un representante con determinante $1$, concretamente $\frac{1}{\sqrt{\det A}} A$. Por tanto, $\operatorname{PGL}_2(\mathbb{C})$ es isomorfo a $\operatorname{PSL}_2(\mathbb{C}) \defeq \sfrac{\operatorname{SL}_2(\mathbb{C})}{\{\pm I\}}$ donde
\[
\operatorname{SL}_2(\mathbb{C}) \defeq \left\{ A \in \operatorname{GL}_2(\mathbb{C}) : \det A = 1 \right\}.
\]
Así, tenemos que $\exhyperref[defn:transformacion-mobius]{transformacion-mobius}{\operatorname{M\ddot{o}b}} \left(\hat{\mathbb{C}}\right) \cong \operatorname{PGL}_2(\mathbb{C}) \cong \operatorname{PSL}_2(\mathbb{C})$.

Para $f \in \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{\mathrm{Aut} \left(\mathbb{D} \right)}$, por el \excref[teo:formula-aut-disco-unidad]{teo-formula-aut-disco-unidad}, $\exists \gamma \in \mathbb{T}, w \in \mathbb{D} : f = \rho_{\gamma} \circ \tau_w$. Normalizando, es fácil ver que, si $A_{\rho_{\gamma}}$ y $A_{\tau_w}$ son las matrices correspondientes a $\rho_{\gamma}$ y $\tau_w$ respectivamente, entonces
\[
[A_{\rho_{\gamma}}] = \left[\begin{pmatrix}
    e^{i \theta} & 0 \\
    0 & e^{-i \theta}
\end{pmatrix}\right] \quad \text{ y }\quad [A_{\tau_w}] = \left[\begin{pmatrix}
    \alpha & \beta \\
    \bar{\beta} & \bar{\alpha}
\end{pmatrix}\right] \quad \text{ en } \quad \operatorname{PSL}_2(\mathbb{C})
\]
donde $\gamma = e^{i\theta}$, $\alpha = \frac{i}{\sqrt{1 - \abs{w}^2}}$ y $\beta = \frac{-iw}{\sqrt{1 - \abs{w}^2}}$. Por tanto, $\mathrm{Aut} \left(\mathbb{D} \right)$ es isomorfo a
\[
\operatorname{PSU}_{1, 1} \left(\C\right) \defeq \left\{ \begin{pmatrix}
    a & b \\
    \bar{b} & \bar{a}
\end{pmatrix} : a, b \in \C \we \abs{a}^2 - \abs{b}^2 = 1 \right\} \big/ \{\pm I\} \subset \operatorname{PSL}_2(\mathbb{C}).
\]
% A su vez, $\operatorname{PSU}_{1, 1} \left(\C\right)$ es isomorfo a $\operatorname{PSL}_2(\mathbb{R})$. %ORDENAR no sé si poner esto o no

%ORDENAR: aquí podría decir que con el \excref[teo:formula-aut-disco-unidad]{teo-formula-aut-disco-unidad} se puede dotar al espacio topológico $\mathbb{T} \times \mathbb{D}$ de una estructura de grupo (y a $\mathrm{Aut} \left(\mathbb{D} \right) \cong \operatorname{PSU}_{1, 1} \left(\C\right)$ de una topología) de manera que se obtiene un grupo de Lie.

\section{Teorema de Schwarz-Pick}

\begin{teo}[Schwarz-Pick]\label{teo:schwarz-pick}
    Sea $f \in \exhyperref[defn:clase-schur]{clase-schur}{\mathscr{S}}$, entonces
    \begin{enumerate}[label=\textnormal{(\arabic*)}]
        \item\label{teo:schwarz-pick:eq1} $\ds \forall w, z \in \mathbb{D} : \abs{\frac{f(w) - f(z)}{1 - \overline{f(w)} f(z)}} \leq \abs{\frac{w - z}{1 - \bar{w} z}}$
        \hspace{2em}\inlineitem\label{teo:schwarz-pick:eq2} $\ds \forall z \in \mathbb{D} : \abs{f'(z)} \leq \frac{1 - \abs{f(z)}^2}{1 - \abs{z}^2}$
    \end{enumerate}

    % \begin{align}
    %     \forall w, z \in \mathbb{D} &:\;\; \abs{\frac{f(z) - f(w)}{1 - \overline{f(w)} f(z)}} \leq \abs{\frac{z - w}{1 - \bar{w} z}} \label{teo:schwarz-pick:eq1} \\
    %     \forall z \in \mathbb{D} &:\;\; \abs{f'(z)} \leq \frac{1 - \abs{f(z)}^2}{1 - \abs{z}^2}. \label{teo:schwarz-pick:eq2}
    % \end{align}

    Además, las siguientes afirmaciones son equivalentes:
    \begin{enu}[2][label=\textnormal{(\roman*)}, ref=\textnormal{(\roman*)}]
        \item\label{teo:schwarz-pick:i} $\exists z \neq w \in \mathbb{D} : $ hay igualdad en \ref{teo:schwarz-pick:eq1}.
        \item\label{teo:schwarz-pick:ii} $\forall z \neq w \in \mathbb{D} : $ hay igualdad en \ref{teo:schwarz-pick:eq1}.
        \item\label{teo:schwarz-pick:iii} $\exists z \in \mathbb{D} : $ hay igualdad en \ref{teo:schwarz-pick:eq2}.
        \item\label{teo:schwarz-pick:iv} $\forall z \in \mathbb{D} : $ hay igualdad en \ref{teo:schwarz-pick:eq2}.
        \item\label{teo:schwarz-pick:v} $f \in \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{\mathrm{Aut} \left(\mathbb{D} \right)}$.
        \item[]
    \end{enu}
\end{teo}
\begin{dem}
    Sea $w \in \mathbb{D}$. Si $\abs{f(w)} = 1$, por el \exhyperref[teo:modulo-maximo]{teo-modulo-maximo}{principio del módulo máximo}, tenemos que $f$ es constante y \ref{teo:schwarz-pick:eq1} y \ref{teo:schwarz-pick:eq2} se cumplen trivialmente. Si, por otro lado, $\abs{f(w)} < 1$, el \exhyperref[teo:modulo-maximo]{teo-modulo-maximo}{principio del módulo máximo} nos dice que $f(\mathbb{D}) \subset \mathbb{D}$.

    Definimos $g_w \defeq \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_{f(w)}} \circ f \circ \tau_w$. Entonces, $g_w \in \mathcal{H} \left(\mathbb{D}\right)$ y $g_w(0) = \tau_{f(w)}(f(w)) = 0$. Por el \exhyperref[lem:schwarz]{lem-schwarz}{lema de Schwarz}, tenemos que $\forall z \in \mathbb{D} : \abs{g_w(z)} \leq \abs{z}$ y $\abs{g_w'(0)} \leq 1$.

    Evaluando en $\tau_w(z)$ en la primera desigualdad, obtenemos \ref{teo:schwarz-pick:eq1}:
    \[
    \abs{g_w(\tau_w(z))} = \abs{\tau_{f(w)}(f(z))} = \abs{\frac{f(z) - f(w)}{1 - \overline{f(w)} f(z)}} \leq \abs{\tau_w(z)} = \abs{\frac{z - w}{1 - \bar{w} z}}.
    \]
    Por otro lado, por la regla de la cadena y el \excref[lem:involucion-disco-unidad-derivada]{lem-involucion-disco-unidad-derivada} en la segunda, llegamos a \ref{teo:schwarz-pick:eq2}:
    \[
    \begin{aligned}
        \abs{g_w'(0)} &= \abs{\tau_{f(w)}'(f(w))} \cdot \abs{f'(w)} \cdot \abs{\tau_w'(0)} \\
        &= \abs{\frac{1}{\abs{f(w)}^2 - 1}} \cdot \abs{f'(w)} \cdot \abs{\abs{w}^2 - 1} \leq 1 \iff \abs{f'(w)} \leq \frac{1 - \abs{f(w)}^2}{1 - \abs{w}^2}.
    \end{aligned}
    \]

    \begin{itemize}[wide, labelindent=0pt]
        \item[$\text{\ref{teo:schwarz-pick:i}} \!\Rightarrow \!\text{\ref{teo:schwarz-pick:v}}$:] Si para $z \neq w$ hay igualdad en \ref{teo:schwarz-pick:eq1}, entonces $\tau_{w}(z) \neq 0$, luego hay igualdad en el \exhyperref[lem:schwarz]{lem-schwarz}{lema de Schwarz} aplicado a $g_w$ para un punto no nulo y, por tanto, $g_w$ es una rotación. Así, $f = \tau_{f(w)} \circ g_w \circ \tau_{w} \in \mathrm{Aut}\left(\mathbb{D}\right)$.

        \item[$\text{\ref{teo:schwarz-pick:iii}} \!\Rightarrow \!\text{\ref{teo:schwarz-pick:v}}$:] Si hay igualdad en \ref{teo:schwarz-pick:eq2} para $w \in \mathbb{D}$, por el \exhyperref[lem:schwarz]{lem-schwarz}{lema de Schwarz} aplicado a $g_w$, tenemos que $\abs{g_w'(0)} = 1$ implica que $g_w$ es una rotación, de donde $f \in \mathrm{Aut}\left(\mathbb{D}\right)$.

        \item[$\text{\ref{teo:schwarz-pick:v}} \!\Rightarrow \!\text{\ref{teo:schwarz-pick:ii}} \!\we \!\text{\ref{teo:schwarz-pick:iv}}$:] Si $f \in \mathrm{Aut}\left(\mathbb{D}\right)$, entonces para cada $w \in \mathbb{D}$ la función $g_w$ es un automorfismo de $\mathbb{D}$ que fija $0$. Por la \excref[obs:aut-disco-unidad-fija-origen-imp-rotacion]{obs-aut-disco-unidad-fija-origen-imp-rotacion}, $g_w$ es una rotación, luego hay igualdad en las dos desigualdades de Schwarz para $g_w$, y al deshacer el cambio de variables se obtiene la igualdad en \ref{teo:schwarz-pick:eq1} para todo $z \neq w$ y en \ref{teo:schwarz-pick:eq2} para todo $z$.

        \item[] Las implicaciones $\text{\ref{teo:schwarz-pick:ii}} \!\Rightarrow \!\text{\ref{teo:schwarz-pick:i}}$ y $\text{\ref{teo:schwarz-pick:iv}} \!\Rightarrow \!\text{\ref{teo:schwarz-pick:iii}}$ son inmediatas.\hfill\qedsymbol
    \end{itemize}
\end{dem}

\begin{obs}
    El \exhyperref[teo:schwarz-pick]{teo-schwarz-pick}{teorema de Schwarz-Pick} se reduce al \exhyperref[lem:schwarz]{lem-schwarz}{Lema de Schwarz} al tomar $w = 0$ y suponer $f(0) = 0$: la desigualdad \ref{teo:schwarz-pick:eq1} da $\abs{f(z)} \leq \abs{z}$ y \ref{teo:schwarz-pick:eq2} da $\abs{f'(0)} \leq 1$.
\end{obs}

\begin{cor}\label{cor:schwarz-pick-involucion-disco-unidad}
    Sea $w \in \mathbb{D}$ y $\tau_w$ la \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{involución del disco unidad} asociada, entonces
    \begin{enumerate}[label=\textnormal{(\arabic*)}]
        % \item $\ds \forall w, z \in \mathbb{D} : \abs{\frac{\tau_w(w) - \tau_w(z)}{1 - \overline{\tau_w(w)} \tau_w(z)}} \leq \abs{\frac{w - z}{1 - \bar{w} z}}$
        \item $\ds \forall w, z \in \mathbb{D} : \abs{\tau_{\tau_w(w)} \left(\tau_w(z)\right)} \leq \abs{\tau_w(z)}$.
        \hspace{1.5em}\inlineitem $\ds \forall z \in \mathbb{D} : \abs{\exhyperref[defn:fn-compleja-derivable-pnt]{fn-compleja-derivable-pnt}{{\tau_w}'}(z)} \leq \frac{1 - \abs{\tau_w(z)}^2}{1 - \abs{z}^2}$
    \end{enumerate}
\end{cor}
\begin{dem}
    Basta con aplicar el \exhyperref[teo:schwarz-pick]{teo-schwarz-pick}{teorema de Schwarz-Pick} a $f = \tau_w \in \exhyperref[defn:clase-schur]{clase-schur}{\mathscr{S}}$.\hfill\qedsymbol
\end{dem}

\subsection{Un problema extremal}

El \exhyperref[teo:schwarz-pick]{teo-schwarz-pick}{teorema de Schwarz-Pick} nos permite resolver problemas extremales relacionados con la clase de Schur. Veamos un ejemplo concreto.

Sean $\alpha, \beta \in \mathbb{D}$ y definamos $\mathcal{A}_{\alpha, \beta} \defeq \{ f \in \exhyperref[defn:clase-schur]{clase-schur}{\mathscr{S}} : f(\alpha) = \beta \}$. Consideremos $f_0 \defeq \tau_\beta \circ \tau_\alpha$. Entonces $f_0 \in \mathcal{H}\left(\mathbb{D} \right)$ y $f_0(\alpha) = \beta$, de modo que $\mathcal{A}_{\alpha, \beta} \neq \varnothing$. Definamos
\[
M = \sup_{f \in \mathcal{A}_{\alpha, \beta}} \abs{f'(\alpha)}.
\]

Sea $f \in \mathcal{A}_{\alpha, \beta}$. Por el \exhyperref[teo:schwarz-pick]{teo-schwarz-pick}{Teorema de Schwarz-Pick}, tenemos que
\[
\abs{f'(\alpha)} \leq \frac{1 - \abs{f(\alpha)}^2}{1 - \abs{\alpha}^2} = \frac{1 - \abs{\beta}^2}{1 - \abs{\alpha}^2}.
\]
Además, para $f = \tau_\beta \circ \tau_{\alpha}$, se satisface la igualdad, luego $M = \frac{1 - \abs{\beta}^2}{1 - \abs{\alpha}^2}$:
\[
\abs{\left(\tau_\beta \circ \tau_{\alpha}\right)'(\alpha)} = \abs{\tau_\beta'(0)} \cdot \abs{\tau_{\alpha}'(\alpha)} \dwexpl{=}{\exref[lem:involucion-disco-unidad-derivada]{lem-involucion-disco-unidad-derivada}} \left(\abs{\beta}^2 - 1\right) \cdot \frac{1}{\abs{\alpha}^2 - 1} = \frac{1 - \abs{\beta}^2}{1 - \abs{\alpha}^2}.
\]

Si $f \in \mathcal{A}_{\alpha, \beta}$ alcanza el supremo, entonces hay igualdad en la segunda desigualdad del \exhyperref[teo:schwarz-pick]{teo-schwarz-pick}{teorema de Schwarz-Pick}, luego $f \in \mathrm{Aut} \left(\mathbb{D} \right)$. Definimos $g \defeq \tau_\beta \circ f \circ \tau_\alpha$. Entonces, $g \in \mathrm{Aut} \left(\mathbb{D} \right)$ y $g(0) = \tau_{\beta} \left(f(\alpha)\right) = \tau_\beta(\beta) = 0$. Por la \excref[obs:aut-disco-unidad-fija-origen-imp-rotacion]{obs-aut-disco-unidad-fija-origen-imp-rotacion}, $\exists \gamma \in \mathbb{T} : g = \rho_\gamma$. Por tanto, al despejar $f$, llegamos a $f = \tau_\beta \circ \rho_\gamma \circ \tau_\alpha$.
