\chapter{Productos finitos de Blaschke}

\section{Definición y propiedades básicas}

\begin{defn}[Producto finito de Blaschke]\label{defn:producto-finito-blaschke}
    Un producto finito de Blaschke de grado $n$ es una función $B \colon \mathbb{D} \to \mathbb{D}$ de la forma
    \[
    B(z) = \gamma \prod_{k=1}^n \frac{z_k - z}{1 - \overline{z_k} z}, 
    \]
    donde $\gamma \in \mathbb{T}$ y $\left\{z_k\right\}_{k=1}^n \subset \mathbb{D}$.
    Un producto finito de Blaschke es mónico si $\gamma = 1$.

    Denotaremos por $\mathscr{B}_n$ al conjunto de productos finitos de Blaschke de grado $n$ y por $\mathscr{B}$ al conjunto de productos finitos de Blaschke, es decir, $\mathscr{B} = \bigcup_{n \in \N} \mathscr{B}_n$.
\end{defn}

Un producto finito de Blaschke es el producto de una constante de módulo $1$ y un número finito de \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{involuciones del disco unidad}. Por el \excref[teo:formula-aut-disco-unidad]{teo-formula-aut-disco-unidad}, esto significa que $B \in \mathscr{B}$ si y sólo si es el producto de un número finito de elementos en \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{$\mathrm{Aut}(\mathbb{D})$}.

Como cada $\exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_{z_k}}$ tiene un cero en $z_k$ y un polo en $\sfrac{1}{\bar{z_k}}$, el producto finito de Blaschke $B$ tiene $n$ ceros en $\left\{z_k\right\}_{k=1}^n \subset \mathbb{D}$ y $n$ polos en $\left\{\sfrac{1}{\bar{z_k}}\right\}_{k=1}^n \subset \widehat{\mathbb{C}} \setminus \bar{\mathbb{D}}$.

Además, $B$ es una \exhyperref[defn:fn-meromorfa]{fn-meromorfa}{función meromorfa} en $\widehat{\mathbb{C}}$ y \exhyperref[defn:fn-holomorfa]{fn-holomorfa}{holomorfa} en $\mathbb{D}$.

\begin{obs}\label{obs:prod-finito-blaschke-clase-schur}
    Por ser un producto de \exhyperref[defn:automorfismo-disco-unidad]{automorfismo-disco-unidad}{automorfismos del disco unidad}, un \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{producto finito de Blaschke} $B$ pertenece a la \exhyperref[defn:clase-schur]{clase-schur}{clase de Schur}, $\mathscr{S}$, y satisface
    \begin{enu}[1][label=(\arabic*), ref=\arabic*]
        \item $\forall z \in \mathbb{D} : \abs{B(z)} < 1$,
        \inlineitem $\forall \zeta \in \mathbb{T} : \abs{B(\zeta)} = 1$,
        \inlineitem $\forall z \in \widehat{\mathbb{C}} \setminus \bar{\mathbb{D}} : \abs{B(z)} > 1$.
    \end{enu}

    Como $\xi \in \mathbb{T}$ implica que $\xi = \sfrac{1}{\bar{\xi}}$ y $B(\xi) \in \mathbb{T}$, tenemos que
    \[
    \forall \zeta \in \mathbb{T} : \ds B(\zeta) = \frac{1}{\bar{B(\zeta)}} = \frac{1}{\bar{B \left(\sfrac{1}{\bar{\zeta}}\right)}}
    \]
    Ahora bien, como $\sfrac{1}{\bar{B \left(\sfrac{1}{\bar{z}}\right)}}$ define una \exhyperref[defn:fn-meromorfa]{fn-meromorfa}{función meromorfa} en $\widehat{\mathbb{C}}$ que coincide con $B$ en $\mathbb{T}$, el \exhyperref[cor:principio-identidad]{cor-principio-identidad}{principio de identidad} nos dice que $\forall z \in \widehat{\mathbb{C}} : B(z) = \sfrac{1}{\bar{B \left(\sfrac{1}{\bar{z}}\right)}}$.
\end{obs}

\subsection{Caracterización de los productos finitos de Blaschke}

\begin{defn}\label{defn:fn-holomorfa-dentro-camino}
    Diremos que una función $f$ es holomorfa dentro de un \exhyperref[defn:camino]{camino}{camino} \exhyperref[defn:curva-cerrada]{curva-cerrada}{cerrado} $\Gamma$ si y sólo si existe un abierto $\Omega$ \exhyperref[defn:conexion-simple]{conexion-simple}{simplemente conexo} tal que $\Gamma \subset \Omega$ y $f \in \exhyperref[defn:fn-holomorfa]{fn-holomorfa}{\mathcal{H}} \left(\Omega\right)$.
\end{defn}

\begin{defn}\label{defn:numero-ceros-fn-holomorfa-camino}
    Usaremos $Z_f(\Gamma)$ para denotar el número de ceros de una función $f$ holomorfa dentro de un camino cerrado $\Gamma$ contando multiplicidades. Formalmente,
    \[
    Z_f(\Gamma) = \!\!\!\!\!\!\!\!\! \sum_{z \in \operatorname{int}(\Gamma) : f(z) = 0} \!\!\!\!\!\!\!\!\! \exhyperref[defn:orden-cero-fn-holomorfa]{orden-cero-fn-holomorfa}{\operatorname{ord}}(f, z).
    \]
\end{defn}

Con esta nueva notación, podemos enunciar de forma sencilla y precisa la siguiente caracterización de los productos finitos de Blaschke:

\begin{teo}\label{teo:carac-prod-finito-blaschke}
    Sea $f \in \exhyperref[defn:fn-holomorfa]{fn-holomorfa}{\mathcal{H}} \left(\mathbb{D}\right) \cap \exhyperref[defn:continuidad]{continuidad}{\mathcal{C}} \left(\bar{\mathbb{D}}\right)$ tal que $\forall \zeta \in \mathbb{T} : \abs{f(\zeta)} = 1$. Entonces, $f$ es un \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{producto finito de Blaschke} de grado $\exhyperref[defn:numero-ceros-fn-holomorfa-camino]{numero-ceros-fn-holomorfa-camino}{Z_f}(\mathbb{T})$.
\end{teo}
\begin{dem}
    Si $f$ tuviese un número infinito de ceros en $\mathbb{D}$, como $\bar{\mathbb{D}}$ es compacto, existiría una sucesión de ceros de $f$ que converge a un punto $z_0 \in \bar{\mathbb{D}}$. Si $z_0 \in \mathbb{D}$, por el \exhyperref[teo:ceros-aislados]{teo-ceros-aislados}{principio de los ceros aislados}, $f \equiv 0$ en $\mathbb{D}$ y, por continuidad, $f \equiv 0$ en $\bar{\mathbb{D}}$. Si $z_0 \in \mathbb{T}$, por continuidad, $f(z_0)=0$. En cualquier caso, llegaríamos a una contradicción con $\abs{f(\zeta)} = 1$ para todo $\zeta \in \mathbb{T}$. Por tanto, $f$ tiene un número finito de ceros en $\mathbb{D}$, digamos $\left(z_k\right)_{k=1}^n \subset \mathbb{D}$ repetidos con multiplicidad. Consideramos $B = \prod_{k=1}^n \tau_{z_k} \in \mathscr{B}_n$.

    Entonces, la función $g = \sfrac{f}{B}$ es holomorfa en $\mathbb{D} \setminus \left\{z_k\right\}_{k=1}^n$ y, como los ceros de $B$ (y sus \exhyperref[defn:orden-cero-fn-holomorfa]{orden-cero-fn-holomorfa}{órdenes}) coinciden con los de $f$, las singularidades de $g$ en $\left\{z_k\right\}_{k=1}^n$ son \exhyperref[defn:singularidad-evitable]{singularidad-evitable}{evitables}, dando lugar a una función holomorfa en $\mathbb{D}$ (\excref[teo:singularidad-evitable-riemann]{teo-singularidad-evitable-riemann}) que no se anula en $\mathbb{D}$.

    Además, para $\zeta \in \mathbb{T}$, tenemos que $\abs{g(\zeta)} = \sfrac{\abs{f(\zeta)}}{\abs{B(\zeta)}} = 1$, luego por el \exhyperref[teo:modulo-maximo]{teo-modulo-maximo}{principio del módulo máximo}, $\forall z \in \mathbb{D} : \abs{g(z)} \leq 1$. Como lo mismo se cumple para $\sfrac{1}{g}$, concluimos que $\forall z \in \mathbb{D} : \abs{g(z)} = 1$. Por el \exhyperref[teo:apl-abierta-compleja]{teo-apl-abierta-compleja}{teorema de la aplicación abierta}, $g$ es constante y de módulo $1$, es decir, $\exists \gamma \in \mathbb{T} : g \equiv \gamma$. Por tanto, $f(z) = \gamma B(z)$.\hfill\qedsymbol
\end{dem}

\section{Unicidad y no unicidad}

\begin{lem}\label{lem:prod-finitos-blaschke-igualdad-n-puntos}
    Sean $B_1, B_2 \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_n}$ mónicos tales que $B_1(w_k) = B_2(w_k)$ para $n$ puntos distintos $\left\{w_k\right\}_{k=1}^n \subset \mathbb{D}$. Entonces, existe $\xi \in \mathbb{T}$ tal que $B_1(\xi) = B_2(\xi)$.
    % \[
    % \big(\exists \left\{w_k\right\}_{k=1}^n \subset \mathbb{D} \text{ distintos} : \forall k \in \N_n : B_1(w_k) = B_2(w_k)\big) \implies \exists \xi \in \mathbb{T} : B_1(\xi) = B_2(\xi). 
    % \]
\end{lem}
\begin{dem}
    Tenemos que $\exists \left\{z_k\right\}_{k=1}^n, \left\{v_k\right\}_{k=1}^n \subset \mathbb{D}$ tales que
    \[
    \begin{aligned}
        B_1(z) = B_2(z) &\iff \prod_{k=1}^n \frac{z_k - z}{1 - \overline{z_k} z} = \prod_{k=1}^n \frac{v_k - z}{1 - \overline{v_k} z} \\
        &\iff \left(\prod_{k=1}^n \frac{z_k - z}{1 - \overline{z_k} z}\right) \bigg/ \left(\prod_{k=1}^n \frac{v_k - z}{1 - \overline{v_k} z}\right) = 1 \\
        &\iff \prod_{k=1}^n \frac{(z_k - z)(1 - \overline{v_k} z)}{(v_k - z)(1 - \overline{z_k} z)} = 1.
    \end{aligned}
    \]

    Para $z \in \mathbb{T}$, esta última igualdad es equivalente a
    \[
    \prod_{k=1}^n \frac{(z_k - z)(\bar{z} - \bar{v_k})}{(v_k - z)(\bar{z} - \bar{z_k})} = 1 \iff \prod_{k=1}^n \left(\frac{z_k - z}{v_k - z}\right) \bigg/\, \bar{\left(\frac{z_k - z}{v_k - z}\right)} = 1.
    \]
    Ahora bien, como $\forall w \in \mathbb{C} \setminus \left\{0\right\} : \ds \sfrac{w}{\bar{w}} = e^{2i \arg(w)}$, basta con probar que
    \[
    \exists \xi \in \mathbb{T} : \arg \left(\prod_{k=1}^n \frac{z_k - \xi}{v_k - \xi}\right) \equiv 0 \mod \pi. 
    \]

    Para ello, definimos $F(z) = \prod_{k=1}^n \frac{z_k - z}{v_k - z}$. Así, $\exists \delta > 0$ tal que $F$ es una función holomorfa y distinta de cero en $\left\{z \in \C : \abs{z} > 1 - \delta\right\}$. Además, tomamos $F(\infty) = \lim_{\abs{z} \to \infty} F(z) = 1$. Por tanto, $H(z) = \log F(z)$ define una función holomorfa en $\left\{z \in \C : \abs{z} > 1 - \delta\right\}$ con $H(\infty) = 0$.%REVISAR:

    Ahora consideramos $h(z) = \Im \left(H \left(\sfrac{1}{z}\right)\right)$ definida en $\left\{z \in \C : \abs{z} < \sfrac{1}{1 - \delta}\right\}$. Entonces, $h$ es armónica en un entorno abierto de $\bar{\mathbb{D}}$. Por la propiedad del valor medio para funciones armónicas, tenemos que
    \[
    h(0) = \frac{1}{2\pi} \int_0^{2\pi} h\left(e^{i\theta}\right) \odif{\theta} \implies 0 = \frac{1}{2\pi} \int_0^{2\pi} h \left(e^{i\theta}\right) \odif{\theta}.
    \]
    Como $h$ es continua y real, por el teorema de los valores intermedios, $\exists \zeta \in \mathbb{T} : h(\zeta) = 0$ (en caso contrario, $h$ sería estrictamente positiva o negativa en $\mathbb{T}$, lo que contradice la integral anterior). Ahora bien, como $\Im H$ es una rama de $\arg F$ (i.e. $F = \abs{F} e^{i \Im H}$) y las ramas de $\arg$ difieren en múltiplos de $2\pi$, el lema queda demostrado.\hfill\qedsymbol
\end{dem}

\begin{teo}\label{teo:unicidad-prod-finitos-blaschke}
    Sean $B_1, B_2 \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_n}$ mónicos tales que $B_1(w_k) = B_2(w_k)$ para $n$ puntos distintos $\left\{w_k\right\}_{k=1}^n \subset \mathbb{D}$. Entonces, $B_1 = B_2$ en $\mathbb{D}$.
    % \[
    % \Big(\exists \left\{w_k\right\}_{k=1}^n \subset \mathbb{D} \text{ distintos} : \forall k \in \N_n : B_1(w_k) = B_2(w_k)\Big) \implies B_1 = B_2 \text{ en } \mathbb{D}.
    % \]
\end{teo}
\begin{dem}
    Definimos la función $R$ mediante $R(z) = \frac{B_1(z)}{B_2(z)}$. Como $B_1, B_2 \in \mathscr{B}$, por la \excref[obs:prod-finito-blaschke-clase-schur]{obs-prod-finito-blaschke-clase-schur}, $\forall \zeta \in \mathbb{T} : \abs{R(\zeta)} = 1$ y, además,
    \[
    \Big(\forall j \in \{1, 2\} : B_j(z)\cdot \bar{B_j \left(\sfrac{1}{\bar{z}}\right)} = 1\Big) \implies R(z) \cdot \bar{R \left(\sfrac{1}{\bar{z}}\right)} = 1.
    \]
    Por hipótesis, $\forall k \in \N_n : R(w_k) = 1$, luego tenemos que $\forall k \in \N_n : R\left(\sfrac{1}{\bar{w_k}}\right) = 1$.
    
    Por el \excref[lem:prod-finitos-blaschke-igualdad-n-puntos]{lem-prod-finitos-blaschke-igualdad-n-puntos}, $\exists \xi \in \mathbb{T} : R(\xi) = 1$. Por tanto, tenemos al menos $2n + 1$ puntos distintos en los que $R$ toma el valor $1$. Como $R$ es una función racional de grado a lo sumo $2n$, concluimos que $R \equiv 1$ en $\widehat{\mathbb{C}}$, es decir, $B_1 = B_2$ en $\mathbb{D}$.\hfill\qedsymbol
\end{dem}

\begin{ejem}%[de necesidad de ser mónicos en el {\excref[teo:unicidad-prod-finitos-blaschke]{teo-unicidad-prod-finitos-blaschke}}]\mbox{}
    Sean $\left\{z_k\right\}_{k=1}^n \subset \mathbb{D} \setminus \{0\}$ distintos, definimos:
    \[
    B_1(z) = i \cdot \prod_{k=1}^n \frac{z_k - z}{1 - \overline{z_k} z} \quad \we \quad B_2(z) = \frac{iz_n - z}{1 + i \bar{z_n} z} \cdot \prod_{k=1}^{n-1} \frac{z_k - z}{1 - \overline{z_k} z}.
    \]
    Entonces, $B_1, B_2 \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_n}$ coinciden en los $n-1$ puntos distintos $\left\{z_k\right\}_{k=1}^{n-1}$ y en $0$, ya que
    \[
    B_1(0) = i \cdot \prod_{k=1}^n z_k  \quad \we \quad B_2(0) = iz_n \cdot \prod_{k=1}^{n-1} z_k = i \cdot \prod_{k=1}^n z_k. 
    \]
    Es decir, cumplen todas las hipótesis del \excref[teo:unicidad-prod-finitos-blaschke]{teo-unicidad-prod-finitos-blaschke} salvo la de ser mónicos. Sin embargo, $B_1 \neq B_2$ en $\mathbb{D}$ ya que si $B_1 = B_2$ en $\mathbb{D}$, tendríamos que
    \[
    \forall z \in \mathbb{D} : 1 = \frac{B_1(z)}{B_2(z)} = i \frac{\exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_{z_n}}(z)}{\exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_{iz_n}}(z)} \implies \tau_{iz_n} = i \tau_{z_n},
    \]
    luego, como $\tau_{iz_n} \circ \tau_{i z_n} = \operatorname{id}$ (\excref[lem:involucion-disco-unidad]{lem-involucion-disco-unidad}), obtendríamos que $\forall z \in \mathbb{D} : \tau_{z_n} \circ \tau_{i z_n} (z) = i z$, lo cual es falso porque $\tau_{z_n} \circ \tau_{i z_n} (i z_n) = \tau_{z_n}(0) = z_n \neq i^2 z_n = -z_n$.

    Por lo tanto, la condición de ser mónicos en el \excref[teo:unicidad-prod-finitos-blaschke]{teo-unicidad-prod-finitos-blaschke} es necesaria para garantizar la unicidad.
\end{ejem}

\begin{defn}\label{defn:valencia-fn-holomorfa}
    Decimos que una función $f \in \exhyperref[defn:fn-holomorfa]{fn-holomorfa}{\mathcal{H}} \left(\Omega\right)$ es $n$-valente en el abierto $\Omega \subset \C$ si para cada $w \in f(\Omega)$, la ecuación $f(z) = w$ tiene exactamente $n$ soluciones en $\Omega$ (contando multiplicidades). Formalmente, $f$ es $n$-valente en $\Omega$ si, y sólo si,
    \[
    \forall w \in f(\Omega) : v_f(w) \defeq \!\!\!\!\!\! \sum_{z \in f^{-1}(\{w\})} \!\!\!\!\!\! \exhyperref[defn:orden-cero-fn-holomorfa]{orden-cero-fn-holomorfa}{\operatorname{ord}}(f-w, z) = n.
    \]
\end{defn}

\begin{teo}\label{teo:n-valencia-prod-finito-blaschke-disco-unidad}
    Todo \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{producto finito de Blaschke de grado $n$} es \exhyperref[defn:valencia-fn-holomorfa]{valencia-fn-holomorfa}{$n$-valente} en $\mathbb{D}$.
\end{teo}
\begin{dem}
    Sea $B \in \mathscr{B}_n$, entonces, existe un $R > 0$ tal que $\bar{\mathbb{D}} \subset D(0, R)$ tal que $B \in \mathcal{H}\left(D(0, R)\right)$. Sea $w \in \mathbb{D}$, si tomamos $f(z) = B(z)$ y $g(z) = B(z) - w$, como $\forall \zeta \in \mathbb{T} : \abs{B(\zeta)} = 1$ y $\abs{w} < 1$, se tiene que
    \[
    \forall \zeta \in \mathbb{T} : \abs{f(\zeta) - g(\zeta)} = \abs{B(\zeta) - (B(\zeta) - w)} = \abs{w} < 1 = \abs{B(\zeta)} = \abs{f(\zeta)}.
    \]
    Por tanto, por \exhyperref[teo:rouche]{teo-rouche}{teorema de Rouché}, $\exhyperref[defn:numero-ceros-fn-holomorfa-camino]{numero-ceros-fn-holomorfa-camino}{Z_f}(\mathbb{T}) = \exhyperref[defn:numero-ceros-fn-holomorfa-camino]{numero-ceros-fn-holomorfa-camino}{Z_g}(\mathbb{T})$. Como $B$ tiene $n$ ceros en $\mathbb{D}$, concluimos que $B(z) - w$ también tiene $n$ ceros en $\mathbb{D}$, es decir, la ecuación $B(z) = w$ tiene $n$ soluciones en $\mathbb{D}$ (contando multiplicidades).\hfill\qedsymbol
\end{dem}

\begin{defn}[Derivada logarítmica]\label{defn:derivada-logaritmica}
    Sea $f$ una función meromorfa en un abierto $\Omega \subset \C$. La derivada logarítmica de $f$ es la función $\mathcal{D} f \colon \Omega \setminus \{z : f(z) = 0\} \to \C$ definida por
    \[
    \mathcal{D} f (z) \defeq \frac{f'(z)}{f(z)}.
    \]
\end{defn}

\begin{lem}\label{lem:derivada-logaritmica-prod-finito-blaschke}
    Sea $B = \gamma \prod_{k=1}^n \exhyperref[defn:involucion-disco-unidad]{involucion-disco-unidad}{\tau_{z_k}} \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_n}$, entonces su \exhyperref[defn:derivada-logaritmica]{derivada-logaritmica}{derivada logarítmica} es
    \[
    \forall z \in \mathbb{D} \setminus \{z_k\}_{k=1}^n : \frac{B'(z)}{B(z)} = \sum_{k=1}^n \frac{\abs{z_k}^2 - 1}{\left(1 - \overline{z_k} z\right) (z_k - z)}. 
    \]
\end{lem}
\begin{dem}
    Basta con aplicar el \excref[lem:involucion-disco-unidad-derivada]{lem-involucion-disco-unidad-derivada}:
    \[
    \frac{B'(z)}{B(z)} = \sum_{k=1}^n \frac{\tau_{z_k}'(z)}{\tau_{z_k}(z)} = \sum_{k=1}^n \frac{\abs{z_k}^2 - 1}{{\left(1 - \overline{z_k} z\right)}^2} \cdot \frac{1 - \overline{z_k} z}{z_k - z} = \sum_{k=1}^n \frac{\abs{z_k}^2 - 1}{\left(1 - \overline{z_k} z\right) (z_k - z)}.
    \]

    \vspace{-\baselineskip}
    \vspace{-\belowdisplayskip}\hfill\qedsymbol
\end{dem}

\begin{lem}\label{lem:derivada-prod-finito-blaschke-no-nula-toro}
    Sea $B$ un \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{producto finito de Blaschke}, entonces $\forall \zeta \in \mathbb{T} : B'(\zeta) \neq 0$.
\end{lem}
\begin{dem}
    Sea $\zeta \in \mathbb{T}$, entonces por el \excref[lem:derivada-logaritmica-prod-finito-blaschke]{lem-derivada-logaritmica-prod-finito-blaschke}, tenemos que
    \[
    \zeta \frac{B'(\zeta)}{B(\zeta)} = \sum_{k=1}^n \frac{\abs{z_k}^2 - 1}{\bar{\zeta} (1 - \overline{z_k} \zeta)(z_k - \zeta)} = \sum_{k=1}^n \frac{\abs{z_k}^2 - 1}{\bar{\left(\zeta - z_k\right)} (\zeta - z_k)} = \sum_{k=1}^n \frac{\abs{z_k}^2 - 1}{\abs{\zeta - z_k}^2} < 0.
    \]
    Como $\abs{B(\zeta)} = 1$ y $\abs{\zeta} = 1$, concluimos que $\ds \abs{B'(\zeta)} = \sum_{k=1}^n \frac{1 - \abs{z_k}^2}{\abs{\zeta - z_k}^2} > 0$.\hfill\qedsymbol
\end{dem}

\begin{obs}
    El \excref[lem:derivada-prod-finito-blaschke-no-nula-toro]{lem-derivada-prod-finito-blaschke-no-nula-toro} implica que un \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{producto finito de Blaschke} no puede alcanzar ningún valor unimodular en $\mathbb{T}$ con multiplicidad mayor que uno. 
    
    En efecto, sea $\gamma \in \mathbb{T}$ y supongamos que $\exists \zeta_0 \in \mathbb{T} : B(\zeta_0) = \gamma$. Si $\zeta_0$ fuera un cero de multiplicidad $m \geq 2$ de la función $h(z) = B(z) - \gamma$, entonces tendríamos que $h'(\zeta_0) = B'(\zeta_0) = 0$, lo cual contradice que $\abs{B'(\zeta_0)} > 0$.
    
    Por tanto, todas las soluciones de la ecuación $B(z) = \gamma$ en $\mathbb{T}$ (para $\gamma \in \mathbb{T}$) son simples, es decir, todas las soluciones son distintas.
\end{obs}

\begin{cor}\label{cor:derivada-arg-prod-finito-blaschke-toro}
    Sea $B \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}}$, entonces $\ds \forall t \in \R : \odv{}{t} \arg B\left(e^{it}\right) = \abs{B'\left(e^{it}\right)}$.
\end{cor}
\begin{dem}
    Como $\forall t \in \R : B\left(e^{it}\right) \in \mathbb{T}$, podemos escribir $B\left(e^{it}\right) = e^{i \psi(t)}$ para alguna función $\psi \colon \R \to \R$ tal que $\psi(2\pi) - \psi(0) = 2\pi m$ para algún $m \in \Z$. Entonces,
    \[
    i e^{it} B'\left(e^{it}\right) = i \psi'(t) B\left(e^{it}\right) \implies \psi'(t) = e^{it} \frac{B'\left(e^{it}\right)}{B\left(e^{it}\right)}.
    \]

    Aplicando el \excref[lem:derivada-logaritmica-prod-finito-blaschke]{lem-derivada-logaritmica-prod-finito-blaschke} en $z = e^{it}$, tenemos que
    \[
    \psi'(t) = e^{it} \sum_{k=1}^n \frac{\abs{z_k}^2 - 1}{\left(1 - \overline{z_k} e^{it}\right) (z_k - e^{it})} = \sum_{k=1}^n \frac{\abs{z_k}^2 - 1}{\bar{\left(e^{it} - z_k\right)} (e^{it} - z_k)} = \sum_{k=1}^n \frac{1 - \abs{z_k}^2}{\abs{e^{it} - z_k}^2}.
    \]
    Entonces, por la demostración del \excref[lem:derivada-prod-finito-blaschke-no-nula-toro]{lem-derivada-prod-finito-blaschke-no-nula-toro}, $\abs{B'\left(e^{it}\right)} = \sum_{k=1}^n \frac{1 - \abs{z_k}^2}{\abs{e^{it} - z_k}^2}$,
    y por tanto $\psi'(t) = \abs{B'\left(e^{it}\right)}$.
    Como $\psi = \arg B\left(e^{it}\right)$ (salvo suma de una constante $2\pi m$), concluimos la igualdad del enunciado.\hfill\qedsymbol
\end{dem}

\begin{obs}\label{obs:n-valencia-prod-finito-blaschke-toro-exterior}
    La \exhyperref[defn:valencia-fn-holomorfa]{defn-valencia-fn-holomorfa}{definición de $n$-valencia} dada al comienzo de esta sección, se aplica a funciones holomorfas en abiertos. Si queremos usar el mismo lenguaje para funciones meromorfas y conjuntos arbitrarios, podemos extenderla asi: sea $E \subset \widehat{\mathbb{C}}$ y sea $f$ meromorfa en un entorno abierto de $E$. Diremos que $f$ es $n$-valente en $E$ si y solo si, para cada $w \in f(E)$, la ecuación $f(z)=w$ tiene exactamente $n$ soluciones en $E$, contadas con multiplicidad. Para $w \in \mathbb{C}$, la multiplicidad de una solución $z_0 \in E$ es $\operatorname{ord}(f-w,z_0)$; para $w=\infty$, la multiplicidad de un polo $z_0 \in E$ es $\operatorname{ord}(1/f,z_0)$.

    Con esta extension, todo $B \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_n}$ es $n$-valente tanto en $\widehat{\mathbb{C}} \setminus \bar{\mathbb{D}}$ como en $\mathbb{T}$. En el exterior del disco, para cada $w$ con $\abs{w}>1$, la ecuacion $B(z)=w$ tiene exactamente $n$ soluciones en $\widehat{\mathbb{C}} \setminus \bar{\mathbb{D}}$, contando multiplicidades. En la circunferencia, para cada $\gamma \in \mathbb{T}$, la ecuacion $B(z)=\gamma$ tiene exactamente $n$ soluciones en $\mathbb{T}$, todas simples.
\end{obs}

\section{Productos de Blaschke como funciones racionales}

\begin{defn}[Polinomio conjugado reverso]\label{defn:polinomio-conjugado-reverso}
    Dado $n > 0$ y $P \in \C[z]$ tal que $\deg P \leq n$, el \emph{polinomio conjugado reverso} de $P$ respecto a $n$ se denota por $P^{\#_n}$ y se define mediante 
    \[
    \forall z \in \C : P^{\#_n}(z) = z^n \bar{P \left(\sfrac{1}{\bar{z}}\right)}.
    \]
\end{defn}

\begin{obs}
    Sea $n > 0$ y $P \in \C[z] : \deg P \leq n$. Entonces:
    \begin{enumerate}[label=(\arabic*)]
        \item Si $P(z) = \sum_{k=0}^n \alpha_k z^k$, entonces $P^{\#_n}(z) = \sum_{k=0}^n \bar{\alpha_{n-k}} z^k$.
        \item Si $P(z) = 1$, entonces $P^{\#_n}(z) = z^n$ y si $P(z) = z^n$, entonces $P^{\#_n}(z) = 1$.
    \end{enumerate}
\end{obs}

\begin{lem}\label{lem:polinomio-conjugado-reverso}
    Sea $n > 0$ y $P \in \C[z] : \deg P \leq n$. Entonces:
    \begin{enumerate}[label=\textnormal{(\arabic*)}]
        \item $\deg{P^{\exhyperref[defn:polinomio-conjugado-reverso]{polinomio-conjugado-reverso}{\#_n}}} \leq n$ y la igualdad se cumple si y sólo si $P(0) \neq 0$.
        \item $\left(P^{\exhyperref[defn:polinomio-conjugado-reverso]{polinomio-conjugado-reverso}{\#_n}}\right)^{\exhyperref[defn:polinomio-conjugado-reverso]{polinomio-conjugado-reverso}{\#_n}} = P$.
        \item $\forall z \in \C \setminus \{0\} : \big(P(z) = 0 \iff P^{\exhyperref[defn:polinomio-conjugado-reverso]{polinomio-conjugado-reverso}{\#_n}} \left(\sfrac{1}{\bar{z}}\right) = 0\big)$.
    \end{enumerate}
\end{lem}

\begin{dem}
    Existen $\left\{\alpha_k\right\}_{k=0}^n \subset \C$ tales que $P(z) = \sum_{k=0}^n \alpha_k z^k$. Entonces,
    \begin{enumerate}[label=(\arabic*), wide, labelindent=0pt]
        \item Por definición, $P^{\#_n}(z) = \sum_{k=0}^n \bar{\alpha_{n-k}} z^k$. Por tanto, $\deg P^{\#_n} \leq n$ y la igualdad se cumple si y sólo si $\bar{\alpha_0} \neq 0$, es decir, si $P(0) = \alpha_0 \neq 0$.
        \item $\ds (P^{\#_n})^{\#_n}(z) = z^n \overline{P^{\#_n}\left(\sfrac{1}{\bar{z}}\right)} = z^n \overline{\frac{1}{\bar{z}^n} \bar{P(z)}} = z^n \cdot \frac{1}{z^n} \cdot P(z) = P(z)$.
        \item Para $z \neq 0$, $P(z) = 0 \iff \bar{P(z)} = 0 \iff z^n \bar{P(z)} = 0 \iff P^{\#_n}\left(\sfrac{1}{\bar{z}}\right) = 0$.\hfill\qedsymbol
    \end{enumerate}
\end{dem}

\begin{prop}\label{prop:conjugado-reverso-producto}
    Sean $P, Q \in \C[z]$ tales que $\deg P \leq n, \deg Q \leq m$. Entonces,
    \[
    \left(P \cdot Q\right)^{\exhyperref[defn:polinomio-conjugado-reverso]{polinomio-conjugado-reverso}{\#_{n+m}}} = P^{\exhyperref[defn:polinomio-conjugado-reverso]{polinomio-conjugado-reverso}{\#_n}} \cdot Q^{\exhyperref[defn:polinomio-conjugado-reverso]{polinomio-conjugado-reverso}{\#_m}}.
    \]
\end{prop}
\begin{dem}
    Se sigue directamente de la definición.\hfill\qedsymbol
\end{dem}

\begin{teo}\label{teo:carac-prod-finito-blaschke-fn-racional}
    Sea $f$ una función racional de grado $n > 0$, entonces
    \[
    f \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_n} \iff \exists P \in \C[z] \text{ de grado } n : \left\{z \in \C : P(z) = 0\right\} \subset \mathbb{D} \we f = \frac{P}{P^{\exhyperref[defn:polinomio-conjugado-reverso]{polinomio-conjugado-reverso}{\#_n}}}.
    \]
\end{teo}
\begin{dem}
    Veamos ambas implicaciones por separado.
    \begin{itemize}[wide, labelindent=0pt]
        \item[$\boxed{\Rightarrow}$] Sea $f$ un producto de Blaschke de grado $n$. Entonces, $\exists \left\{z_k\right\}_{k=1}^n \subset \mathbb{D}, \gamma \in \mathbb{T} : $
        \[
        \forall z \in \mathbb{D} : f(z) = \gamma \prod_{k=1}^n \frac{z_k - z}{1 - \overline{z_k} z} = e^{t_0i} \cdot \frac{\prod_{k=1}^n (z_k - z)}{\prod_{k=1}^n (1 - \overline{z_k} z)}
        \]
        para algún $t_0 \in \R$. Definimos ahora el polinomio $P$ mediante
        \[
        \forall z \in \C : P(z) = (-1)^n e^{t_0i/2} \cdot \prod_{k=1}^n (z_k - z).
        \]
        Entonces, $P$ es un polinomio de grado $n$ cuyos ceros están en $\mathbb{D}$ y
        \[
        P^{\#_n}(z) = z^n \bar{P \left(\sfrac{1}{\bar{z}}\right)} = z^n \cdot e^{-t_0i/2} \cdot \prod_{k=1}^n \bar{\left(z_k - \sfrac{1}{\bar{z}}\right)} = e^{-t_0i/2} \cdot \prod_{k=1}^n (1 - \overline{z_k} z).
        \]
        Por tanto, $\forall z \in \mathbb{D} : f(z) = \sfrac{P(z)}{P^{\#_n}(z)}$.

        \item[$\boxed{\Leftarrow}$] Supongamos que $f = \sfrac{P}{P^{\#_n}}$ para $P \in \C[z]$ de grado $n$ y dado por
        \[
        P(z) = \alpha_n z^n + \alpha_{n-1} z^{n-1} + \cdots + \alpha_1 z + \alpha_0,
        \]
        cuyos ceros están en $\mathbb{D}$. Entonces, $\exists \left\{z_k\right\}_{k=1}^n \subset \mathbb{D}: P(z) = (-1)^n \alpha_n \prod_{k=1}^n (z_k - z)$.
        \[
        \begin{aligned}
            \therefore \quad P^{\#_n}(z) = z^n \bar{P \left(\sfrac{1}{\bar{z}}\right)} = z^n\, \bar{\alpha_n} \prod_{k=1}^n \bar{\left(z_k - \sfrac{1}{\bar{z}}\right)} = \bar{\alpha_n} \prod_{k=1}^n (1 - \overline{z_k} z).
        \end{aligned}
        \]
        Por tanto, $\forall z \in \mathbb{D} : \ds f(z) = \frac{P(z)}{P^{\#_n}(z)} = \frac{\alpha_n}{\bar{\alpha_n}} \cdot \prod_{k=1}^n \frac{z_k - z}{1 - \overline{z_k} z}$ donde $\ds \abs{\frac{\alpha_n}{\bar{\alpha_n}}} = 1$.\hfill\qedsymbol
    \end{itemize}
\end{dem}

%ORDENAR: me ha dicho Dragan que es muy prescindible
\begin{cor}\label{cor:carac-cociente-prod-finito-blaschke-fn-racional}
    Sea $f$ una función racional de grado $n$, entonces
    \[
    \begin{aligned}
        \exists B_1, B_2  \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}} : f = \frac{B_1}{B_2} \iff \exists P \in \C[z] : \big(\forall z \in \mathbb{T} : P(z) \neq 0\big) \we f = \frac{P}{P^{\exhyperref[defn:polinomio-conjugado-reverso]{polinomio-conjugado-reverso}{\#_n}}}.
    \end{aligned}
    \]
    Además, en tal caso, $\deg B_1 + \deg B_2 = n = \deg P$.
\end{cor}
\begin{dem}
    Veamos ambas implicaciones por separado.
    \begin{itemize}[wide, labelindent=0pt]
        \item[$\boxed{\Rightarrow}$] Por el \excref[teo:carac-prod-finito-blaschke-fn-racional]{teo:carac-prod-finito-blaschke-fn-racional}, $\exists Q, R \in \C[z]$ de grados $n_1 \defeq \deg B_1$ y $n_2 \defeq\deg B_2$ respectivamente, cuyos ceros están en $\mathbb{D}$, tales que $B_1 = \sfrac{Q}{Q^{\#_{n_1}}}$ y $B_2 = \sfrac{R}{R^{\#_{n_2}}}$.
        
        Por tanto, si definimos $P \defeq Q \cdot R^{\#_{n_2}}$, $\deg P = \deg Q + \deg R^{\#_{n_2}} \leq n_1 + n_2 = n$ (por el \excref[lem:polinomio-conjugado-reverso]{lem-polinomio-conjugado-reverso}) y los ceros de $P$ están en $\mathbb{D} \cup \C \setminus \bar{\mathbb{D}} = \C \setminus \mathbb{T}$. Además,
        \[
        f = \frac{B_1}{B_2} = \frac{Q / Q^{\#_{n_1}}}{R / R^{\#_{n_2}}} = \frac{Q \cdot R^{\#_{n_2}}}{Q^{\#_{n_1}} \cdot R} \upexpl{=}{\exref[lem:polinomio-conjugado-reverso]{lem-polinomio-conjugado-reverso}}  \frac{Q \cdot R^{\#_{n_2}}}{Q^{\#_{n_1}} \cdot \left(R^{\#_{n_2}}\right)^{\#_{n_2}}} \upexpl{=}{\exref[prop:conjugado-reverso-producto]{prop-conjugado-reverso-producto}} \frac{Q \cdot R^{\#_{n_2}}}{\left(Q \cdot R^{\#_{n_2}}\right)^{\#_{n_1 + n_2}}} = \frac{P}{P^{\#_n}}.
        \]

        % Como $\deg R^{\#_{n_2}} \leq n_2$ y $\deg Q^{\#_{n_1}} \leq n_1$ y se tiene que
        % \[
        % n = \max \left\{\deg \left(Q R^{\#_{n_2}}\right), \deg \left(Q^{\#_{n_1}} R\right)\right\} = \max\left\{n_1 + \deg R^{ \#_{n_2}}, n_2 + \deg Q^{\#_{n_1}}\right\},
        % \]
        % concluimos que $\deg B_1 + \deg B_2 = n_1 + n_2 \leq n$.

        \item[$\boxed{\Leftarrow}$] Sea $f$ una función racional de la forma $f(z) = \sfrac{P(z)}{P^{\#_n}(z)}$ donde $P$ es un polinomio sin ceros en $\mathbb{T}$. Entonces, podemos separar sus $n$ ceros en dos grupos: los $n_1$ que están en $\mathbb{D}$ y los $n_2$ que están en $\C \setminus \bar{\mathbb{D}}$. Por tanto, existen $Q, R$ polinomios de grados $n_1$ y $n_2$ respectivamente, cuyos ceros están en $\mathbb{D}$, tales que $P = Q \cdot R^{\#_{n_2}}$.
        
        Por la \excref[prop:conjugado-reverso-producto]{prop-conjugado-reverso-producto}, se tiene que
        \[
        f = \frac{P}{P^{\#_n}} = \frac{Q \cdot R^{\#_{n_2}}}{\left(Q \cdot R^{\#_{n_2}}\right)^{\#_n}} = \frac{Q \cdot R^{\#_{n_2}}}{Q^{\#_{n_1}} \cdot R} = \frac{Q}{Q^{\#_{n_1}}} \bigg/ \frac{R}{R^{\#_{n_2}}}.
        \]
        Luego, por el \excref[teo:carac-prod-finito-blaschke-fn-racional]{teo:carac-prod-finito-blaschke-fn-racional}, existen $B_1$ y $B_2$ productos finitos de Blaschke tales que $\sfrac{Q}{Q^{\#_{n_1}}} = B_1$ y $\sfrac{R}{R^{\#_{n_2}}} = B_2$. Por tanto, $f = \sfrac{B_1}{B_2}$.\hfill\qedsymbol

        % Por último, tenemos que $n = \deg f = \max \left\{\deg P, \deg P^{\#_n}\right\} = \deg P$.\hfill\qedsymbol
    \end{itemize}
\end{dem}

\begin{obs}
    Sea $P$ un polinomio de grado $n$ sin ceros en $\mathbb{T}$. Entonces, por el \excref[cor:carac-cociente-prod-finito-blaschke-fn-racional]{cor-carac-cociente-prod-finito-blaschke-fn-racional},
    $\forall \zeta \in \mathbb{T} : \abs{P(\zeta)} = \abs{P^{\#_n}(\zeta)}$.
\end{obs}

\section{Álgebra del disco}

\begin{defn}[Álgebra del disco unidad]\label{defn:algebra-disco-unidad}
    El conjunto de funciones holomorfas en $\mathbb{D}$ que se extienden de forma continua a $\bar{\mathbb{D}}$ es un álgebra con la suma, el producto de funciones y el producto por escalares. Lo llamamos \emph{álgebra del disco unidad} y se denota por $\mathscr{A}(\mathbb{D})$.
\end{defn}

Por ejemplo, $\mathscr{B} \subset \mathscr{A}(\mathbb{D})$. De hecho, cualquier función racional sin polos en $\bar{\mathbb{D}}$ pertenece al \exhyperref[defn:algebra-disco-unidad]{algebra-disco-unidad}{álgebra del disco}.

\begin{ejem}
    Consideramos $f$ dada por $f(z) = \sqrt{1-z}$ para $z \in \mathbb{D}$. Entonces, tomando la rama principal de la raíz cuadrada, vemos que $f$ es holomorfa en $\mathbb{D}$. Observamos que $f$ se extiende de forma continua a $\bar{\mathbb{D}} \setminus \{1\}$ directamente.

    Para estudiar el límite en $1$, escribimos $z = re^{it}$ con $0 \leq r < 1$. Si $z \to 1$ dentro de $\bar{\mathbb{D}}$, entonces $r \to 1$ y $t \to 0$ (módulo $2\pi$). Como
    \[
    \abs{1-z}^2 = \abs{1-re^{it}}^2 = 1+r^2-2r\cos t \to 0,
    \]
    tenemos que $\abs{\sqrt{1-z}} = \sqrt{\abs{1-z}} \to 0$. Definiendo $f(1)=0$, obtenemos una extensión continua a $\bar{\mathbb{D}}$, es decir, $f \in \mathscr{A}(\mathbb{D})$.
\end{ejem}

El siguiente resultado nos permitirá caracterizar los productos finitos de Blaschke como funciones del álgebra del disco.

\begin{teo}[Fatou]\label{teo:fatou-carac-prod-finito-blaschke-algebra-disco}
    Sea $f \in \exhyperref[defn:fn-holomorfa]{fn-holomorfa}{\mathcal{H}}\left(\mathbb{D}\right)$ tal que $\ds \exhyperref[defn:convergencia]{convergencia}{\lim_{\abs{z} \to 1^-}} \abs{f(z)} = 1$. Entonces, $f \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}}$.
\end{teo}
\begin{dem}
    Como $\abs{f(z)} \xrightarrow{\abs{z} \to 1^-} 1$, tenemos que
    \[
    \forall \varepsilon > 0 : \exists \delta \in (0, 1) : \forall z \in \mathbb{D} : \left(1 - \delta < \abs{z} \implies \abs{\abs{f(z)} - 1} < \varepsilon\right).
    \]
    En particular, tomando $\varepsilon < 1$, $\exists r \in (0, 1) : \forall z \in \mathbb{D} : \left(r < \abs{z} < 1 \implies \abs{f(z)} > 0\right)$, es decir, $f$ no se anula en $\left\{z \in \mathbb{D} : r < \abs{z} < 1\right\}$. Entonces, $f$ tiene todos sus ceros en el compacto $\bar{D(0, r)} = \left\{z \in \C : \abs{z} \leq r\right\}$, luego por el \excref[cor:fn-holomorfa-compacto-imp-finitud-ceros]{cor-fn-holomorfa-compacto-imp-finitud-ceros}, $f$ tiene un número finito de ceros en $\mathbb{D}$.
    
    Sea $B$ el producto finito de Blaschke con los mismos ceros que $f$ (contando multiplicidades). Entonces, $\sfrac{f}{B}$ y $\sfrac{B}{f}$ son holomorfas en $\mathbb{D}$ y satisfacen que
    \[
    \lim_{\abs{z} \to 1^-} \abs{\frac{f(z)}{B(z)}} = \lim_{\abs{z} \to 1^-} \abs{\frac{B(z)}{f(z)}} = 1.
    \]
    Por el principio del módulo máximo, $\forall z \in \mathbb{D} : \abs{\sfrac{f(z)}{B(z)}}, \abs{\sfrac{B(z)}{f(z)}} \leq 1$, luego $\abs{\sfrac{f}{B}}$ es constante igual a $1$. Por la \excref[prop:modulo-constante-imp-constante]{prop-modulo-constante-imp-constante}, $\exists \gamma \in \mathbb{T} : f = \gamma B$, es decir, $f$ es un producto finito de Blaschke.\hfill\qedsymbol
\end{dem}

\begin{cor}\label{cor:carac-prod-finito-blaschke-algebra-disco}
    Si $f \in \exhyperref[defn:algebra-disco-unidad]{algebra-disco-unidad}{\mathscr{A}(\mathbb{D})}$ y $f(\mathbb{T}) \subseteq \mathbb{T}$, entonces $f$ es un \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{producto de Blaschke}.
\end{cor}
\begin{dem}
    Por hipótesis, $\abs{f(\zeta)} = 1$ para todo $\zeta \in \mathbb{T}$. Como $f$ es continua en el compacto $\overline{\mathbb{D}}$, $f$ es uniformemente continua en $\overline{\mathbb{D}}$. Entonces, dado $\varepsilon > 0$, existe $\eta > 0$ tal que $\forall z,w \in \overline{\mathbb{D}}:\ \abs{z-w}<\eta \implies \abs{f(z)-f(w)}<\varepsilon$.
    Si $z \in \mathbb{D}$ y $1-\eta < \abs{z} < 1$, tomando $\zeta = z/\abs{z} \in \mathbb{T}$ se tiene $\abs{z-\zeta}=1-\abs{z}<\eta$, luego
    \[
    \abs{\abs{f(z)}-1}=\abs{\abs{f(z)}-\abs{f(\zeta)}}\leq \abs{f(z)-f(\zeta)}<\varepsilon.
    \]
    Por tanto, $\abs{f(z)} \to 1$ cuando $\abs{z} \to 1^-$ y aplicamos el \excref[teo:fatou-carac-prod-finito-blaschke-algebra-disco]{teo-fatou-carac-prod-finito-blaschke-algebra-disco}.\hfill\qedsymbol
\end{dem}

\begin{cor}\label{cor:carac-cociente-prod-finito-blaschke-algebra-disco-meromorfa}
    Sea $f$ una \exhyperref[defn:fn-meromorfa]{fn-meromorfa}{función meromorfa} en $\mathbb{D}$ tal que $f \in \exhyperref[defn:continuidad]{continuidad}{\mathcal{C}} \left(\bar{\mathbb{D}}\right)$ y $f(\mathbb{T}) \subseteq \mathbb{T}$. Entonces, $f$ es el cociente de dos \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{productos finitos de Blaschke}.
\end{cor}
\begin{dem}
    Como $f$ es continua en el compacto $\bar{\mathbb{D}}$, $f$ tiene un número finito de polos en $\mathbb{D}$. Sea $B_1$ un producto finito de Blaschke cuyos ceros coinciden con los polos de $f$ (contando multiplicidades). Entonces, $B_2 \defeq B_1 f$ es holomorfa en $\mathbb{D}$ y se extiende de forma continua a $\bar{\mathbb{D}}$. Además, $\forall \zeta \in \mathbb{T} : \abs{B_2(\zeta)} = \abs{B_1(\zeta) f(\zeta)} = 1$. Por el \excref[cor:carac-prod-finito-blaschke-algebra-disco]{cor-carac-prod-finito-blaschke-algebra-disco}, $B_2$ es un producto finito de Blaschke. Por tanto, $f = \sfrac{B_2}{B_1}$ es el cociente de dos productos finitos de Blaschke.\hfill\qedsymbol
\end{dem}

\section{Composición de productos finitos de Blaschke}

Los resultados anteriores nos permiten demostrar que $\mathscr{B}$ es cerrado para la composición de funciones de forma sencilla, evitándonos un razonamiento puramente algebraico, que sería más largo y tedioso.

\begin{lem}\label{lem:prod-finito-blaschke-composicion-involucion}
    Sea $B \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_n}$ y $w \in \mathbb{D}$. Entonces, $\tau_w \circ B, B \circ \tau_w \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_n}$.
\end{lem}
\begin{dem}
    Sabemos que $\tau_w \circ B \in \mathcal{H}(\mathbb{D}) \cap \mathcal{C}\left(\bar{\mathbb{D}}\right)$ y que $\forall \zeta \in \mathbb{T} : \abs{\tau_w \circ B(\zeta)} = 1$. Por tanto, el \cref{teo:fatou-carac-prod-finito-blaschke-algebra-disco} asegura que $\tau_w \circ B$ es un producto finito de Blaschke. Además, como $\tau_w \circ B(z) = 0 \iff B(z) = w$, por el \cref{teo:n-valencia-prod-finito-blaschke-disco-unidad}, la ecuación $\tau_w \circ B(z) = 0$ tiene exactamente $n$ soluciones en $\mathbb{D}$ (contando multiplicidades). Por tanto, $\tau_w \circ B$ es un producto finito de Blaschke de grado $n$. 
    
    Análogamente, el \cref{teo:fatou-carac-prod-finito-blaschke-algebra-disco} asegura que $B \circ \tau_w$ es un producto finito de Blaschke. Además, como $\tau_w$ es biyectiva, por el \cref{teo:n-valencia-prod-finito-blaschke-disco-unidad}, existen $n$ soluciones en $\mathbb{D}$ para la ecuación $B \circ \tau_w(z) = 0$. Por tanto, $B \circ \tau_w \in \mathscr{B}_n$.\hfill\qedsymbol
\end{dem}

\begin{obs}\label{obs:prod-finito-blaschke-composicion-automorfismo-disco-unidad}
    Sea $B \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_n}$. Entonces, como $\forall \gamma \in \mathbb{T} : \rho_\gamma \circ B, B \circ \rho_\gamma \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_n}$, el \excref[teo:formula-aut-disco-unidad]{teo-formula-aut-disco-unidad} y el \excref[lem:prod-finito-blaschke-composicion-involucion]{lem-prod-finito-blaschke-composicion-involucion} aseguran que $\forall \varphi \in \mathrm{Aut}(\mathbb{D}) : \varphi \circ B, B \circ \varphi \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_n}$.
\end{obs}

\begin{teo}\label{teo:composicion-prod-finito-blaschke}
    Sean $B_1 \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_{n_1}}$ y $B_2 \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_{n_2}}$. Entonces, $B_1 \circ B_2 \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}_{n_1 n_2}}$.
\end{teo}
\begin{dem}
    Por definición, $\exists \left\{z_k\right\}_{k=1}^{n_1} \subset \mathbb{D}, \gamma \in \mathbb{T}$ tales que $B_1 = \rho_\gamma \circ \prod_{k=1}^{n_1} \tau_{z_k}$.
    Entonces, $B_1 \circ B_2 = \rho_\gamma \circ \prod_{k=1}^{n_1} \left(\tau_{z_k} \circ B_2\right)$. Por el \excref[lem:prod-finito-blaschke-composicion-involucion]{lem-prod-finito-blaschke-composicion-involucion}, cada uno de los factores $\tau_{z_k} \circ B_2$ es un producto finito de Blaschke de grado $n_2$. Por tanto, $B_1 \circ B_2$ es un producto finito de Blaschke de grado $n_1 n_2$.\hfill\qedsymbol
\end{dem}

\section{Caracterización de $\mathscr{B}$ en $\mathscr{S}$}

Podemos concluir este capítulo con una caracterización de los productos finitos de Blaschke dentro de la clase de Schur. En concreto, veremos que los productos finitos de Blaschke son exactamente las funciones $n$-valentes de la clase de Schur.

\begin{teo}%[]\label{teo:}
    Sea $f \in \exhyperref[defn:clase-schur]{clase-schur}{\mathscr{S}}$ una \exhyperref[defn:valencia-fn-holomorfa]{valencia-fn-holomorfa}{función $n$-valente} en $\mathbb{D}$. Entonces, $f \in \exhyperref[defn:producto-finito-blaschke]{producto-finito-blaschke}{\mathscr{B}}$.
\end{teo}
\begin{dem}
    Por el \cref{teo:fatou-carac-prod-finito-blaschke-algebra-disco}, basta ver que $\lim_{\abs{z} \to 1^-} \abs{f(z)} = 1$. Suponemos por contradicción que $\exists \left(z_m\right)_{m\in \N} \subset \mathbb{D}$ tal que $\abs{z_m} \to 1 \we \neg \left(\abs{f(z_m)} \to 1\right)$. Entonces, existe $r \in (0, 1)$ y una subsucesión $\left(z_{m_j}\right)_{j \in \N}$ tal que $\forall j \in \N : \abs{f(z_{m_j})} \leq r$. Como $\left\{z \in \mathbb{D} : \abs{z} \leq r\right\}$ es compacto, $\exists \big(z_{m_{j_k}}\big)_{k \in \N}$ subsucesión de $\left(z_{m_j}\right)_{j \in \N}$ y $w_0 \in \mathbb{D}$ tales que $f(z_{m_{j_k}}) \xrightarrow{k \to \infty} w_0$. La denotaremos por $\left(z_k\right)_{k \in \N}$.

    Sin pérdida de generalidad, podemos suponer que $w_0 = 0$. En caso contrario, tomamos $g \defeq \tau_{w_0} \circ f$. Entonces, $g \in \mathscr{S}$, $g$ es $n$-valente en $\mathbb{D}$, $g(z_k) \xrightarrow{k \to \infty} 0$ y, si $g \in \mathscr{B}$, entonces $f = \tau_{w_0}^{-1} \circ g \in \mathscr{B}$ por el \excref[lem:prod-finito-blaschke-composicion-involucion]{lem-prod-finito-blaschke-composicion-involucion}.

    Como $f$ es $n$-valente, $f(z) = 0$ tiene $n$ soluciones en $\mathbb{D}$, digamos $\left\{a_\ell\right\}_{\ell=1}^s \subset \mathbb{D}$ con multiplicidades $\left\{m_\ell\right\}_{\ell=1}^s$ respectivamente tales que $\sum_{\ell=1}^s m_\ell = n$. Entonces,
    \[
    \forall \ell \in \N_s : \exists g_\ell \in \mathcal{H}\left(\mathbb{D}\right) : \forall z \in D(a_\ell, \varepsilon_\ell) : \bigl(g_\ell(z) \neq 0 \we f(z) = (z - a_\ell)^{m_\ell} g_\ell(z)\bigr).
    \]
    Sin pérdida de generalidad, podemos suponer que los discos $D(a_\ell, \varepsilon_\ell)$ son disjuntos dos a dos y están contenidos en $\mathbb{D}$ (reduciendo $\varepsilon_\ell$ en caso de que fuese necesario).
    
    Como $D(a_\ell, \varepsilon_\ell)$ es simplemente conexo y $g_\ell$ no se anula en él, existe $h_\ell \in \mathcal{H}\left(D(a_\ell, \varepsilon_\ell)\right)$ tal que $h_\ell^{m_\ell} = g_\ell$. Definimos $\phi_\ell(z) \defeq \left(z - a_\ell\right) h_\ell(z)$ para $z \in D(a_\ell, \varepsilon_\ell)$. Entonces, $\forall z \in D(a_\ell, \varepsilon_\ell) : f(z) = \phi_\ell(z)^{m_\ell}$ y, como $\phi_\ell'(a_\ell) = h_\ell(a_\ell) \neq 0$, por el \exhyperref[teo:fn-inversa-holomorfas]{teo-fn-inversa-holomorfas}{teorema de la función inversa}, $\phi_\ell$ es \exhyperref[defn:inyectividad-local]{inyectividad-local}{localmente inyectiva} en $a_\ell$.
    Reduciendo de nuevo $\varepsilon_\ell$ si es necesario, podemos suponer que $\phi_\ell$ es inyectiva en $D(a_\ell, \varepsilon_\ell)$. Entonces, por el \exhyperref[teo:apl-abierta-compleja]{teo-apl-abierta-compleja}{teorema de la aplicación abierta}, $\phi_\ell\left(D(a_\ell, \varepsilon_\ell)\right)$ es abierto, luego $E \defeq \bigcap_{\ell=1}^s \phi_\ell\left(D(a_\ell, \varepsilon_\ell)\right)$ es abierto. Además, como $\phi_\ell(a_\ell) = 0$ para todo $\ell \in \N_s$, tenemos que $0 \in E$.

    Entonces, $\exists \varepsilon > 0 : D(0, \varepsilon) \subset E$ y tomamos $V_\ell \defeq \phi_\ell^{-1}\left(D(0, \varepsilon)\right) \subset D(a_\ell, \varepsilon_\ell)$ para $\ell \in \N_s$. Observamos que $\phi_\ell\big|_{V_\ell}$ es biyectiva sobre $D(0, \varepsilon)$, $\phi_\ell\left(a_\ell\right) = 0$ y los abiertos $V_\ell \subset \mathbb{D}$ son disjuntos dos a dos. Entonces, tomando $\varepsilon' \defeq \min_{\ell \in \N_s} \varepsilon^{m_\ell}$, para cada $\ell \in \N_s$ y todo $w \in D(0, \varepsilon') \setminus \{0\}$, el conjunto $U_{\ell,w} \defeq \left\{u \in \mathbb{C} : u^{m_\ell} = w\right\}$ tiene exactamente $m_\ell$ elementos y satisface $U_{\ell,w} \subset D(0, \varepsilon)$. Por la biyección $\phi_\ell\big|_{V_\ell}$, a cada $u \in U_{\ell,w}$ le corresponde un único $z \in V_\ell$ con $\phi_\ell(z) = u$ y, recíprocamente, de $f(z) = w$ se sigue que $\phi_\ell(z) \in U_{\ell,w}$. Como $f = \phi_\ell^{m_\ell}$ en $V_\ell$, concluimos que $f(z) = w$ tiene exactamente $m_\ell$ soluciones en $V_\ell$.

    Como $V_\ell \subset D(a_\ell, \varepsilon_\ell)$ para todo $\ell \in \N_s$, se tiene
    \[
    \bigcup_{\ell=1}^s V_\ell \subset \bigcup_{\ell=1}^s D(a_\ell, \varepsilon_\ell) \subset D(0, \sigma), \quad \text{donde } \sigma \defeq \max_{\ell \in \N_s}\left(\abs{a_\ell}+\varepsilon_\ell\right) < 1.
    \]
    Como $f(z_k) \xrightarrow{k \to \infty} 0$, existe $k_1 \in \N$ tal que $\forall k \geq k_1 : f(z_k) \in D(0, \varepsilon') \setminus \{0\}$.
    Como $\abs{z_k} \xrightarrow{k \to \infty} 1$ y $\sigma < 1$, existe $k_2 \in \N$ tal que $\forall k \geq k_2 : \abs{z_k} > \sigma$ y, por tanto,
    \[
    \forall k \geq k_2 : z_k \notin \bigcup_{\ell=1}^s V_\ell.
    \]
    Tomamos $k_0 \defeq \max\{k_1, k_2\}$ y la sucesión $\left(w_k\right)_{k\in \N}$ dada por $w_k = f(z_{k_0 + k})$. Fijado $k \in \N$, para cada $\ell \in \N_s$, la ecuación $f(z) = w_k$ tiene exactamente $m_\ell$ soluciones en $V_\ell$.
    Como los abiertos $V_\ell$ son disjuntos dos a dos, estas soluciones son distintas y suman $m_1 + \dots + m_s = n$. Además, $f(z_{k_0+k}) = w_k$ y $z_{k_0+k} \notin \bigcup_{\ell=1}^s V_\ell$, así que obtenemos una solución adicional.
    Por tanto, la ecuación $f(z) = w_k$ tiene al menos $n + 1$ soluciones. Contradicción. \hfill\qedsymbol
\end{dem}
